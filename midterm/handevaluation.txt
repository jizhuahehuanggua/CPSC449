foldr:: (a->b->b)->b->[a]->b
foldr f b [] = b
foldr f b (x:xs) = f x (foldr f b xs)

mord:: Ord a => a -> [a] -> Bool
mord x = foldr (\a b -> (a==x) || ((x>=a) && b)) False 


(||) :: Bool->Bool->Bool
(||) False  b = b
(||) _  _ = True

(&&) :: Bool->Bool->Bool
(&&) True b = b
(&&) _  _ = False
